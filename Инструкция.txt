// базовые настройки
scoop install kubectl

kubectl version --client

mkdir .kube
cd .kube
New-Item config -type file

kubectl cluster-info
kubectl get nodes

scoop install helm
scoop install minikube
minikube start --driver=docker

// настройка namespace
kubectl apply -f namespace.yaml

// настройка postgres
kubectl create configmap postgres-migrations --namespace logistic --from-file=postgres/migrations/
kubectl apply -f postgres/secrets.yaml 

kubectl get storageclass

kubectl apply -f postgres/pvc.yaml 
kubectl apply -f postgres/deployment.yaml
kubectl apply -f postgres/service.yaml

// настройка сертификатов для https (не особо работает)
kubectl apply -f https://github.com/cert-manager/cert-manager/releases/download/v1.14.2/cert-manager.yaml
kubectl apply -f cluster-issuer.yaml

openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout tls.key -out tls.crt -subj "/CN=logistic-api/O=logistic-api"
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout tls.key -out tls.crt -subj "/CN=logistic-client/O=logistic-client"

kubectl create secret tls tls-secret --cert=./tls.crt --key=./tls.key -n logistic

minikube addons enable ingress

// включение zookeper + kafka
kubectl apply -f kafka/zookeper-statefulset.yaml -n logistic
kubectl apply -f kafka/zookeper-service.yaml -n logistic

kubectl apply -f kafka/kafka-deployment.yaml -n logistic
kubectl apply -f kafka/kafka-service.yaml -n logistic

// включение logistic-api
kubectl apply -f api/deployment.yaml
kubectl apply -f api/service.yaml 

kubectl apply -f ingress.yaml
kubectl apply -f ingress-api-test.yaml

// включение logistic-builder
kubectl apply -f builder/deployment.yaml
kubectl apply -f builder/service.yaml 

kubectl apply -f ingress-service-builder.yaml

// включение minio
kubectl apply -f minio/pvc.yaml
kubectl apply -f minio/secrets.yaml
kubectl apply -f minio/configmap.yaml
kubectl apply -f minio/deployment.yaml

// включение logistic-trainer
kubectl apply -f trainer/deployment.yaml
kubectl apply -f trainer/service.yaml 

// включение logistic-payment
kubectl apply -f payment/deployment.yaml
kubectl apply -f payment/service.yaml 

kubectl apply -f ingress-service-payment.yaml

// включение logistic-client
kubectl apply -f web/deployment.yaml
kubectl apply -f web/service.yaml 

kubectl apply -f ingress-client.yaml

// включение nginx (как прокси для ingress контроллеров)
helm repo add ingress-nginx https://kubernetes.github.io/ingress-nginx
helm repo update
helm install ingress-nginx ingress-nginx/ingress-nginx

kubectl apply -f nginx/configmap.yaml 
kubectl apply -f nginx/deployment.yaml  
kubectl apply -f nginx/service.yaml

// applying migrations
kubectl delete configmap postgres-migrations -n logistic
kubectl create configmap postgres-migrations --namespace logistic --from-file=postgres/migrations/
kubectl delete deployment postgres -n logistic
kubectl delete pvc postgres-pvc -n logistic
kubectl apply -f postgres/pvc.yaml
kubectl apply -f postgres/deployment.yaml
kubectl logs -f deployment/postgres -n logistic

// update service versions
kubectl rollout restart deployment/logistic-api -n logistic
kubectl apply -f api/deployment.yaml -n logistic

// restart or scale deployments
kubectl scale deployments logistic-api --replicas=0 -n logistic
kubectl scale deployments logistic-builder --replicas=0 -n logistic

kubectl rollout restart deployment -n logistic logistic-api
kubectl rollout restart deployment -n logistic logistic-builder 

// manual health checks
curl.exe -k https://logistic-api:443/hc
curl.exe -k https://logistic-builder:443/hc
curl.exe -k https://logistic-web:43/
kubectl port-forward deployment/postgres 5433:5432 -n logistic

minikube service logistic-client --url -n logistic

# Исправление работы Kafka
kubectl delete deployment kafka -n logistic

# Очистить данные ZooKeeper
kubectl exec -it zookeeper-0 -n logistic -- bash -c "rm -rf /var/lib/zookeeper/data/version-2/*"

# Перезапустить ZooKeeper
kubectl delete pod zookeeper-0 -n logistic

# Подождать, пока ZooKeeper запустится
kubectl wait --for=condition=ready pod zookeeper-0 -n logistic --timeout=60s
